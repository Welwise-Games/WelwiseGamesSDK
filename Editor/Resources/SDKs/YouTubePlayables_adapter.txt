(function() {
    function sendUnityMessage(handler, message) {
        var interval = setInterval(function() {
            if (window.unityInstance) {
                clearInterval(interval);
                window.unityInstance.SendMessage('PluginRuntime', handler, message || '');
            }
        }, 50);
    }

    window.__sdk_adapter = {
        GetModules: function() {
            return [
                "Advertisement",
                "PlayerData",
                "Environment",
                "Analytics"
            ];
        },

        Init: function() {
            if (typeof ytgame === 'undefined') {
                sendUnityMessage('HandleInitError', 'YouTube Playables SDK not loaded');
                return;
            }

            if (typeof ytgame.system !== 'undefined') {
                window.__yt_audio_enabled = ytgame.system.isAudioEnabled();

                ytgame.system.onAudioEnabledChange(function(isEnabled) {
                    window.__yt_audio_enabled = isEnabled;
                });

                ytgame.system.onPause(function() {
                    sendUnityMessage('HandleGamePause', 'true');
                });
                ytgame.system.onResume(function() {
                    sendUnityMessage('HandleGamePause', 'false');
                });
            }

            sendUnityMessage('HandleInitSuccess', '');
        },

        GetPlayerData: function() {
            if (typeof ytgame === 'undefined' || !ytgame.game) {
                sendUnityMessage('HandleGetDataError', 'YouTube Playables SDK not available');
                return;
            }

            var inYT = typeof ytgame.IN_PLAYABLES_ENV !== 'undefined' && ytgame.IN_PLAYABLES_ENV;

            if (!inYT) {
                var saved = '';
                try { saved = localStorage.getItem('YT_SAVE') || ''; } catch(e) {}
                var universalData = {
                    playerName: '',
                    playerGameData: saved ? [{ identifier: 'cloudSave', value: 's' + saved }] : []
                };
                sendUnityMessage('HandleGetDataSuccess', JSON.stringify(universalData));
                return;
            }

            ytgame.game.loadData()
                .then(function(data) {
                    var universalData = {
                        playerName: '',
                        playerGameData: data ? [{ identifier: 'cloudSave', value: 's' + data }] : []
                    };
                    sendUnityMessage('HandleGetDataSuccess', JSON.stringify(universalData));
                })
                .catch(function(error) {
                    sendUnityMessage('HandleGetDataError',
                        error && error.message ? error.message : 'Failed to load cloud data');
                });
        },

        SetPlayerData: function(jsonData) {
            if (typeof ytgame === 'undefined' || !ytgame.game) {
                sendUnityMessage('HandleSetDataError', 'YouTube Playables SDK not available');
                return;
            }

            try {
                var universalData = JSON.parse(jsonData);
                var savePayload = '';

                if (universalData.playerGameData && universalData.playerGameData.length > 0) {
                    var entry = universalData.playerGameData[0];
                    if (typeof entry.value === 'string' && entry.value.length > 1) {
                        savePayload = entry.value.substring(1);
                    } else {
                        savePayload = String(entry.value);
                    }
                }

                var inYT = typeof ytgame.IN_PLAYABLES_ENV !== 'undefined' && ytgame.IN_PLAYABLES_ENV;

                if (!inYT) {
                    try { localStorage.setItem('YT_SAVE', savePayload); } catch(e) {}
                    sendUnityMessage('HandleSetDataSuccess', '');
                    return;
                }

                var maxBytes = 3 * 1024 * 1024;
                if (new Blob([savePayload]).size >= maxBytes) {
                    sendUnityMessage('HandleSetDataError', 'Cloud save data exceeds 3 MiB limit');
                    return;
                }

                ytgame.game.saveData(savePayload)
                    .then(function() {
                        sendUnityMessage('HandleSetDataSuccess', '');
                    })
                    .catch(function(error) {
                        sendUnityMessage('HandleSetDataError',
                            error && error.message ? error.message : 'Failed to save cloud data');
                    });
            } catch (e) {
                sendUnityMessage('HandleSetDataError', 'Invalid JSON format');
            }
        },

        GetServerTime: function() {
            sendUnityMessage('HandleGetTimeSuccess', String(Date.now()));
        },

        GoToGame: function(gameId) {
            sendUnityMessage('HandleNavigateError', 'Navigation not supported on YouTube Playables');
        },

        GetPlayerId: function() {
            sendUnityMessage('HandleGetPlayerIdSuccess', '');
        },

        GetDeviceType: function() {
            var deviceType = 'desktop';
            var ua = navigator.userAgent || '';
            if (/Android|webOS|iPhone|iPad|iPod|BlackBerry|IEMobile|Opera Mini/i.test(ua)) {
                deviceType = 'mobile';
            } else if (/iPad|Android(?!.*Mobile)/i.test(ua)) {
                deviceType = 'tablet';
            } else if (/SmartTV|GoogleTV|AppTV|TV|SonyDTV/i.test(ua)) {
                deviceType = 'tv';
            }
            sendUnityMessage('HandleGetDeviceTypeSuccess', deviceType);
        },

        GetLanguageCode: function() {
            if (typeof ytgame === 'undefined' || !ytgame.system) {
                sendUnityMessage('HandleGetLanguageCodeSuccess', navigator.language || 'en');
                return;
            }

            ytgame.system.getLanguage()
                .then(function(lang) {
                    sendUnityMessage('HandleGetLanguageCodeSuccess', lang || 'en');
                })
                .catch(function() {
                    sendUnityMessage('HandleGetLanguageCodeSuccess', navigator.language || 'en');
                });
        },

        IsMetaverseSupported: function() {
            sendUnityMessage('HandleIsMetaverseSupportedSuccess', 'false');
        },

        GameReady: function() {
            if (typeof ytgame !== 'undefined' && ytgame.game) {
                ytgame.game.gameReady();
            }
            sendUnityMessage('HandleGameReadySuccess', '');
        },

        GameplayStart: function() {
            sendUnityMessage('HandleGameplayStartSuccess', '');
        },

        GameplayStop: function() {
            sendUnityMessage('HandleGameplayStopSuccess', '');
        },

        ShowInterstitial: function() {
            var inYT = typeof ytgame !== 'undefined' && typeof ytgame.IN_PLAYABLES_ENV !== 'undefined' && ytgame.IN_PLAYABLES_ENV;

            if (!inYT || typeof ytgame === 'undefined' || !ytgame.ads) {
                sendUnityMessage('HandleInterstitialClose', '');
                return;
            }

            ytgame.ads.requestInterstitialAd()
                .then(function() {
                    sendUnityMessage('HandleInterstitialClose', '');
                })
                .catch(function() {
                    sendUnityMessage('HandleInterstitialClose', '');
                });
        },

        ShowRewarded: function() {
            var inYT = typeof ytgame !== 'undefined' && typeof ytgame.IN_PLAYABLES_ENV !== 'undefined' && ytgame.IN_PLAYABLES_ENV;

            if (!inYT || typeof ytgame === 'undefined' || !ytgame.ads) {
                sendUnityMessage('HandleRewardedOpen', '');
                sendUnityMessage('HandleRewardedClose', '');
                return;
            }

            sendUnityMessage('HandleRewardedOpen', '');

            ytgame.ads.requestRewardedAd('default-reward')
                .then(function(isRewardEarned) {
                    if (isRewardEarned) {
                        sendUnityMessage('HandleRewardedRewarded', '');
                    }
                    sendUnityMessage('HandleRewardedClose', '');
                })
                .catch(function() {
                    sendUnityMessage('HandleRewardedClose', '');
                });
        },

        PaymentsInit: function() {
            sendUnityMessage('HandlePaymentsInitError', 'Payments not supported on YouTube Playables');
        },

        PaymentsGetCatalog: function() {
            sendUnityMessage('HandlePaymentsGetCatalogError', 'Payments not supported on YouTube Playables');
        },

        PaymentsGetPurchases: function() {
            sendUnityMessage('HandlePaymentsGetPurchasesError', 'Payments not supported on YouTube Playables');
        },

        PaymentsPurchase: function(productId, developerPayload) {
            sendUnityMessage('HandlePaymentsPurchaseError', 'Payments not supported on YouTube Playables');
        },

        PaymentsConsume: function(purchaseToken) {
            sendUnityMessage('HandlePaymentsConsumeError', 'Payments not supported on YouTube Playables');
        }
    };
})();
