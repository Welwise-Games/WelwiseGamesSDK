const FALLBACK_SDK_URL = 'https://welwise-games.online/sdk';

fetch('./config.json')
    .then(res => res.text())
    .then(text => {
        try {
            const config = JSON.parse(text);
            return config.SDK_URL || FALLBACK_SDK_URL;
        } catch (e) {
            console.warn('config.json parse error, using fallback SDK URL:', e.message);
            return FALLBACK_SDK_URL;
        }
    })
    .catch(err => {
        console.warn('config.json load error, using fallback SDK URL:', err.message);
        return FALLBACK_SDK_URL;
    })
    .then(sdkUrl => {
        return new Promise((resolve, reject) => {
            const script = document.createElement('script');
            script.src = sdkUrl;
            script.onload = resolve;
            script.onerror = reject;
            document.head.appendChild(script);
        });
    })
    .then(() => {
        if (!window.WelwiseGames || typeof window.WelwiseGames.init !== 'function') {
            throw new Error('Welwise SDK not properly loaded');
        }

        window.__sdk_adapter = {
            GetModules: function() {
                return [
                    "Advertisement",
                    "Environment",
                    "PlatformNavigation",
                    "PlayerData",
                    "GameData",
                    "MetaverseData",
                    "Payments"
                ];
            },

            Init: function() {
                const initPromise = WelwiseGames.init()
                .then(sdk => {
                    window.__wg_sdk = sdk;
                    return sdk;
                });

                const sendUnityMessage = (handler, message = '') => {
                    const checkInterval = setInterval(() => {
                        if (window.unityInstance) {
                            clearInterval(checkInterval);
                            window.unityInstance.SendMessage('PluginRuntime', handler, message);
                        }
                    }, 100);
                };

                initPromise
                    .then(() => sendUnityMessage('HandleInitSuccess'))
                    .catch(error => {
                        const errorMessage = error.message || String(error);
                        sendUnityMessage('HandleInitError', errorMessage);
                    });
            },

            GetPlayerData: function() {
                if (!window.__wg_sdk) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage('PluginRuntime', 'HandleGetDataError', 'SDK not initialized');
                    }
                    return;
                }

                window.__wg_sdk.player.getData()
                    .then(data => {
                        const universalData = {
                            playerName: data.playerName || '',
                            playerGameData: data.playerGameData || []
                        };

                        if (window.unityInstance) {
                            window.unityInstance.SendMessage(
                                'PluginRuntime',
                                'HandleGetDataSuccess',
                                JSON.stringify(universalData)
                            );
                        }
                    })
                    .catch(error => {
                        if (window.unityInstance) {
                            window.unityInstance.SendMessage(
                                'PluginRuntime',
                                'HandleGetDataError',
                                error.message || String(error)
                            );
                        }
                    });
            },

            SetPlayerData: function(jsonData) {
                if (!window.__wg_sdk) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage('PluginRuntime', 'HandleSetDataError', 'SDK not initialized');
                    }
                    return;
                }

                try {
                    const data = JSON.parse(jsonData);
                    window.__wg_sdk.player.setData(data)
                        .then(() => {
                            if (window.unityInstance) {
                                window.unityInstance.SendMessage('PluginRuntime', 'HandleSetDataSuccess', '');
                            }
                        })
                        .catch(error => {
                            if (window.unityInstance) {
                                window.unityInstance.SendMessage(
                                    'PluginRuntime',
                                    'HandleSetDataError',
                                    error.message || String(error)
                                );
                            }
                        });
                } catch (e) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage('PluginRuntime', 'HandleSetDataError', 'Invalid JSON format');
                    }
                }
            },

            GetServerTime: function() {
                if (!window.__wg_sdk) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage('PluginRuntime', 'HandleGetTimeError', 'SDK not initialized');
                    }
                    return;
                }

                window.__wg_sdk.serverTime()
                    .then(timestamp => {
                        if (window.unityInstance) {
                            window.unityInstance.SendMessage('PluginRuntime', 'HandleGetTimeSuccess', String(timestamp));
                        }
                    })
                    .catch(error => {
                        if (window.unityInstance) {
                            window.unityInstance.SendMessage(
                                'PluginRuntime',
                                'HandleGetTimeError',
                                error.message || String(error)
                            );
                        }
                    });
            },

            GoToGame: function(gameId) {

                    if (window.unityInstance) {
                        window.unityInstance.SendMessage('PluginRuntime', 'HandleNavigateError', 'SDK not initialized');
                    }
                    return;
                }

                    .then(() => {
                        if (window.unityInstance) {
                            window.unityInstance.SendMessage('PluginRuntime', 'HandleNavigateSuccess', '');
                        }
                    })
                    .catch(error => {
                        if (window.unityInstance) {
                            window.unityInstance.SendMessage(
                                'PluginRuntime',
                                'HandleNavigateError',
                                error.message || String(error)
                            );
                        }
                    });
            },

            GetPlayerId: function() {
                if (!window.__wg_sdk) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage('PluginRuntime', 'HandleGetPlayerIdError', 'SDK not initialized');
                    }
                    return;
                }
                const playerId = window.__wg_sdk.Environment.PlayerId || '';
                if (window.unityInstance) {
                    window.unityInstance.SendMessage('PluginRuntime', 'HandleGetPlayerIdSuccess', playerId);
                }
            },

            GetDeviceType: function() {
                if (!window.__wg_sdk) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage('PluginRuntime', 'HandleGetDeviceTypeError', 'SDK not initialized');
                    }
                    return;

                    window.unityInstance.SendMessage('PluginRuntime', 'HandleGetDeviceTypeSuccess', deviceType);
                }
            },

            GetLanguageCode: function() {
                if (!window.__wg_sdk) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage('PluginRuntime', 'HandleGetLanguageCodeError', 'SDK not initialized');
                    }
                    return;
                }
                const languageCode = window.__wg_sdk.Environment.LanguageCode || '';
                if (window.unityInstance) {
                    window.unityInstance.SendMessage('PluginRuntime', 'HandleGetLanguageCodeSuccess', languageCode);
                }
            },

            GetMetaversePlayerData: function() {
                if (!window.__wg_sdk) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage('PluginRuntime', 'HandleGetMetaverseDataError', 'SDK not initialized');
                    }
                    return;
                }

                window.__wg_sdk.metaversePlayer.getData()
                    .then(data => {
                        if (window.unityInstance) {
                            window.unityInstance.SendMessage(
                                'PluginRuntime',
                                'HandleGetMetaverseDataSuccess',
                                JSON.stringify(data)
                            );
                        }
                    })
                    .catch(error => {
                        if (window.unityInstance) {
                            window.unityInstance.SendMessage(
                                'PluginRuntime',
                                'HandleGetMetaverseDataError',
                                error.message || String(error)
                            );
                        }
                    });
            },

            SetMetaversePlayerData: function(jsonData) {
                if (!window.__wg_sdk) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage('PluginRuntime', 'HandleSetMetaverseDataError', 'SDK not initialized');
                    }
                    return;
                }

                try {
                    const data = JSON.parse(jsonData);
                    window.__wg_sdk.metaversePlayer.setData(data)
                        .then(() => {
                            if (window.unityInstance) {
                                window.unityInstance.SendMessage('PluginRuntime', 'HandleSetMetaverseDataSuccess', '');
                            }
                        })
                        .catch(error => {
                            if (window.unityInstance) {
                                window.unityInstance.SendMessage(
                                    'PluginRuntime',
                                    'HandleSetMetaverseDataError',
                                    error.message || String(error)
                                );
                            }
                        });
                } catch (e) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage('PluginRuntime', 'HandleSetMetaverseDataError', 'Invalid JSON format');
                    }
                }
            },

            IsMetaverseSupported: function() {
                if (!window.__wg_sdk) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage('PluginRuntime', 'HandleIsMetaverseSupportedError', 'SDK not initialized');
                    }
                    return;
                }

                try {
                    const supported = window.__wg_sdk.isMetaverseSupported();
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage(
                            'PluginRuntime',
                            'HandleIsMetaverseSupportedSuccess',
                            supported ? 'true' : 'false'
                        );
                    }
                } catch (e) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage(
                            'PluginRuntime',
                            'HandleIsMetaverseSupportedError',
                            e.message || String(e)
                        );
                    }
                }
            },

            GetCombinedPlayerData: function() {
                if (!window.__wg_sdk) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage('PluginRuntime', 'HandleGetCombinedDataError', 'SDK not initialized');
                    }
                    return;
                }

                window.__wg_sdk.metaversePlayer.getGameData()
                    .then(data => {
                        if (window.unityInstance) {
                            window.unityInstance.SendMessage(
                                'PluginRuntime',
                                'HandleGetCombinedDataSuccess',
                                JSON.stringify(data)
                            );
                        }
                    })
                    .catch(error => {
                        if (window.unityInstance) {
                            window.unityInstance.SendMessage(
                                'PluginRuntime',
                                'HandleGetCombinedDataError',
                                error.message || String(error)
                            );
                        }
                    });
            },

            SetCombinedPlayerData: function(jsonData) {
                if (!window.__wg_sdk) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage('PluginRuntime', 'HandleSetCombinedDataError', 'SDK not initialized');
                    }
                    return;
                }

                try {
                    const data = JSON.parse(jsonData);
                    window.__wg_sdk.metaversePlayer.setGameData(data)
                        .then(() => {
                            if (window.unityInstance) {
                                window.unityInstance.SendMessage('PluginRuntime', 'HandleSetCombinedDataSuccess', '');
                            }
                        })
                        .catch(error => {
                            if (window.unityInstance) {
                                window.unityInstance.SendMessage(
                                    'PluginRuntime',
                                    'HandleSetCombinedDataError',
                                    error.message || String(error)
                                );
                            }
                        });
                } catch (e) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage('PluginRuntime', 'HandleSetCombinedDataError', 'Invalid JSON format');
                    }
                }
            },

            GameReady: function() {
                if (window.unityInstance) {
                    window.unityInstance.SendMessage('PluginRuntime', 'HandleGameReadySuccess', '');
                }
            },

            GameplayStart: function() {
                if (window.unityInstance) {
                    window.unityInstance.SendMessage('PluginRuntime', 'HandleGameplayStartSuccess', '');
                }
            },

            GameplayStop: function() {
                if (window.unityInstance) {
                    window.unityInstance.SendMessage('PluginRuntime', 'HandleGameplayStopSuccess', '');
                }
            },

            ShowInterstitial: function() {
                if (!window.__wg_sdk) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage('PluginRuntime', 'HandleInterstitialError', 'SDK not initialized');
                    }
                    return;
                }

                try {
                    window.__wg_sdk.AdvManager.showMidgame({
                        callbacks: {
                            onOpen: () => {
                                if (window.unityInstance) {
                                    window.unityInstance.SendMessage('PluginRuntime', 'HandleInterstitialOpen', '');
                                }
                            },
                            onClose: () => {
                                if (window.unityInstance) {
                                    window.unityInstance.SendMessage('PluginRuntime', 'HandleInterstitialClose', '');
                                }
                            },
                            onError: (e) => {
                                if (window.unityInstance) {
                                    window.unityInstance.SendMessage(
                                        'PluginRuntime',
                                        'HandleInterstitialError',
                                        e.message || String(e)
                                    );
                                }
                            }
                        }
                    });
                } catch (e) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage(
                            'PluginRuntime',
                            'HandleInterstitialError',
                            e.message || String(e)
                        );
                    }
                }
            },

            ShowRewarded: function() {
                if (!window.__wg_sdk) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage('PluginRuntime', 'HandleRewardedError', 'SDK not initialized');
                    }
                    return;
                }

                try {
                    window.__wg_sdk.AdvManager.showRewarded({
                        callbacks: {
                            onOpen: () => {
                                if (window.unityInstance) {
                                    window.unityInstance.SendMessage('PluginRuntime', 'HandleRewardedOpen', '');
                                }
                            },
                            onRewarded: () => {
                                if (window.unityInstance) {
                                    window.unityInstance.SendMessage('PluginRuntime', 'HandleRewardedRewarded', '');
                                }
                            },
                            onClose: () => {
                                if (window.unityInstance) {
                                    window.unityInstance.SendMessage('PluginRuntime', 'HandleRewardedClose', '');
                                }
                            },
                            onError: (e) => {
                                if (window.unityInstance) {
                                    window.unityInstance.SendMessage(
                                        'PluginRuntime',
                                        'HandleRewardedError',
                                        e.message || String(e)
                                    );
                                }
                            }
                        }
                    });
                } catch (e) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage(
                            'PluginRuntime',
                            'HandleRewardedError',
                            e.message || String(e)
                        );
                    }
                }
            },

            // ======== Payments — адаптация Welwise к эталону ========

            // PaymentsInit: purchases доступен сразу после sdk.init()
            PaymentsInit: function() {
                if (!window.__wg_sdk) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage('PluginRuntime', 'HandlePaymentsInitError', 'SDK not initialized');
                    }
                    return;
                }
                if (window.unityInstance) {
                    window.unityInstance.SendMessage('PluginRuntime', 'HandlePaymentsInitSuccess', '');
                }
            },

            // PaymentsGetCatalog: у Welwise нет каталога — товары на бэкенде.
            // Возвращаем пустой массив, чтобы эталон не падал.
            PaymentsGetCatalog: function() {
                if (window.unityInstance) {
                    window.unityInstance.SendMessage('PluginRuntime', 'HandlePaymentsGetCatalogSuccess', '[]');
                }
            },

            // PaymentsGetPurchases → getPendingPurchases()
            // Маппим поля pending в эталонный формат Purchase:
            //   productID  ← itemId
            //   purchaseToken ← purchaseId  (используется потом в Consume/confirm)
            //   developerPayload ← quantity + debitedCoins (JSON)
            //   signature ← transactionDateTime
            PaymentsGetPurchases: function() {
                if (!window.__wg_sdk) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage('PluginRuntime', 'HandlePaymentsGetPurchasesError', 'SDK not initialized');
                    }
                    return;
                }

                window.__wg_sdk.purchases.getPendingPurchases()
                    .then(pending => {
                        const mapped = (pending || []).map(p => ({
                            productID: p.itemId,
                            purchaseToken: p.purchaseId,
                            developerPayload: JSON.stringify({
                                quantity: p.quantity,
                                debitedCoins: p.debitedCoins,
                                transactionDateTime: p.transactionDateTime
                            }),
                            signature: p.transactionDateTime || ''
                        }));

                        if (window.unityInstance) {
                            window.unityInstance.SendMessage(
                                'PluginRuntime',
                                'HandlePaymentsGetPurchasesSuccess',
                                JSON.stringify(mapped)
                            );
                        }
                    })
                    .catch(error => {
                        if (window.unityInstance) {
                            window.unityInstance.SendMessage(
                                'PluginRuntime',
                                'HandlePaymentsGetPurchasesError',
                                error.message || String(error)
                            );
                        }
                    });
            },

            // PaymentsPurchase(productId, developerPayload) → sdk.purchases.purchase(itemId, ...)
            // developerPayload может содержать quantity как JSON: {"quantity":5}
            // Результат маппится в эталонный Purchase
            PaymentsPurchase: function(productId, developerPayload) {
                if (!window.__wg_sdk) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage('PluginRuntime', 'HandlePaymentsPurchaseError', 'SDK not initialized');
                    }
                    return;
                }

                var quantity = 1;
                if (developerPayload) {
                    try {
                        var payload = JSON.parse(developerPayload);
                        if (payload.quantity && typeof payload.quantity === 'number') {
                            quantity = payload.quantity;
                        }
                    } catch(e) {
                        // developerPayload не JSON — используем quantity=1
                    }
                }

                try {
                    window.__wg_sdk.purchases.purchase(productId, {
                        quantity: quantity,
                        callbacks: {
                            onSuccess: (data) => {
                                var result = {
                                    productID: data.itemId,
                                    purchaseToken: data.purchaseId,
                                    developerPayload: JSON.stringify({
                                        quantity: data.quantity,
                                        debitedCoins: data.debitedCoins
                                    }),
                                    signature: ''
                                };

                                if (window.unityInstance) {
                                    window.unityInstance.SendMessage(
                                        'PluginRuntime',
                                        'HandlePaymentsPurchaseSuccess',
                                        JSON.stringify(result)
                                    );
                                }
                            },
                            onError: (reason, error, purchaseId) => {
                                var msg = (reason || 'INTERNAL_ERROR') + ': ' +
                                    (error ? (error.message || String(error)) : 'Unknown error');
                                if (purchaseId) {
                                    msg += ' [purchaseId=' + purchaseId + ']';
                                }

                                if (window.unityInstance) {
                                    window.unityInstance.SendMessage(
                                        'PluginRuntime',
                                        'HandlePaymentsPurchaseError',
                                        msg
                                    );
                                }
                            }
                        }
                    });
                } catch (e) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage(
                            'PluginRuntime',
                            'HandlePaymentsPurchaseError',
                            e.message || String(e)
                        );
                    }
                }
            },

            // PaymentsConsume(purchaseToken) → sdk.purchases.confirmPurchase(purchaseId)
            // purchaseToken в эталоне = purchaseId в Welwise
            PaymentsConsume: function(purchaseToken) {
                if (!window.__wg_sdk) {
                    if (window.unityInstance) {
                        window.unityInstance.SendMessage('PluginRuntime', 'HandlePaymentsConsumeError', 'SDK not initialized');
                    }
                    return;
                }

                window.__wg_sdk.purchases.confirmPurchase(purchaseToken)
                    .then(() => {
                        if (window.unityInstance) {
                            window.unityInstance.SendMessage(
                                'PluginRuntime',
                                'HandlePaymentsConsumeSuccess',
                                purchaseToken
                            );
                        }
                    })
                    .catch(error => {
                        if (window.unityInstance) {
                            window.unityInstance.SendMessage(
                                'PluginRuntime',
                                'HandlePaymentsConsumeError',
                                error.message || String(error)
                            );
                        }
                    });
            }
        };
    })
    .catch(err => {
        console.error('SDK initialization failed:', err);
        if (window.unityInstance) {
            window.unityInstance.SendMessage(
                'PluginRuntime',
                'HandleInitError',
                err.message || String(err)
            );
        }
    });
